export const Preview = ({ children }) => {
  const childArray = Array.isArray(children) ? children : [children];
  const preview = childArray[0];
  const code = childArray[1];
  
  return (
    <div className="my-6 overflow-hidden rounded-xl border">
      <div className="flex min-h-[350px] items-center justify-center p-10">
        {preview}
      </div>
      {code && <div className="[&_figure]:m-0! [&_figure]:rounded-none! [&_figure]:border-0! [&_figure]:border-t! [&_figure]:dark:border-t-neutral-800! ">{code}</div>}
    </div>
  );
}

<Preview>
{props.children}
