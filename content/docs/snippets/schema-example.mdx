```tsx
// [!code word:Input]
// [!code word:Textarea]
// [!code word:Switch]
Input('title')
  .label('Title')
  .required()
  .placeholder('Enter the article title')

Textarea('subtitle')
  .label('Subtitle')
  .maxLength(100)
  .placeholder('Enter the article subtitle')
  .defaultValue('A content management system for developers')

Switch('published')
  .label('Published')
  .description('Is the article published?')
```
